
@startuml

Aldeano..|>Jugador
Lobo..|>Jugador
Cazador..|>Aldeano
Caballero..|>Aldeano
Vidente..|>Aldeano
AldeanoComun..|>Aldeano
Angel..|>Aldeano

interface Jugador{
+getId():int
+morir():void
+clone():Jugador
}

class Lobo{
-cabello:String
-arma:String
-ropa:String
-aldea:Aldea
-id:int
-nickname:String
+clone():Lobo
+getId():int
+getAldea():Aldea
+getNickname():String
}

class Aldeano{
-cabello:String
-arma:String
-ropa:String
-aldea:Aldea
-id:int
-nickname:String
+getId():int
+getAldea():Aldea
+getNickname():String
}

class Cazador{
-static uniqueinstance:Cazador
+static getinstanceof()
-Cazador()
+setEsbirro():void
}

class Vidente{
-static uniqueinstance:Vidente
+static getinstanceof()

}

class Caballero{
-static uniqueinstance:Caballero
+static getinstanceof()

}

class Angel{
-static uniqueinstance:Angel
+static getinstanceof()

}

}

}

class MagiaAngel{

}
